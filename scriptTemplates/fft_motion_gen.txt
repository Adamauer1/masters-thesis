public FFTMotion SetFFTMotions(Flame[] flames, String propertyPath,  double amplitude, int fftChannel, double offset){
  FFTMotion motion = new FFTMotion();
  for (int i = 0; i < flames.length; i++){
     motion.getMotionLinks().add(new MotionLink( new FlamePropertyPath(flames[i], propertyPath)));
  }
  motion.setAmplitude(amplitude);
  motion.setFftChannel(fftChannel);
  motion.setOffset(offset);
  return motion;
}
